DatabaseConnector.cs

  • Database Connection and Query Execution:
    • Establishes an ODBC connection using connection strings from the configuration.
    • Uses a retry mechanism to handle transient connection errors with multiple attempts.
  • Progress Bar Updates:
    • Pbar2 indicates an ongoing connection as indeterminate.
    • Pbar3 provides determinate progress based on the total row count fetched from the database.
  • Data Fetching:
    • Executes a query to fetch the total row count for progress tracking.
    • Retrieves the data in batches to avoid memory overload, adding rows to the DataTable while periodically updating the progress bar.
  • Thread-Safe UI Updates:
    • Updates progress bars using Dispatcher.Invoke() to ensure that UI changes are thread-safe, since the data fetching is done asynchronously in a non-UI thread.
using System.Data;
using System.Data.Odbc;
using System.Diagnostics;
using System.Windows.Controls;
using System.Configuration;
using System.Text.RegularExpressions;

public class DatabaseConnector
{
    private string odbcConnectionString;
    private const int MaxRetryAttempts = 3;
    private const int RetryDelayMilliseconds = 1000;
    private ProgressBar Pbar2;
    private ProgressBar Pbar3;

    // Constructor to initialize the progress bars and set the connection string from configuration
    public DatabaseConnector(ProgressBar pbar2, ProgressBar pbar3)
    {
        Pbar2 = pbar2;
        Pbar3 = pbar3;
        odbcConnectionString = ConfigurationManager.ConnectionStrings["MaximoDB"].ConnectionString;
    }

    // Asynchronous method to fetch data from an ODBC database
    public async Task<DataTable> FetchDataODBCAsync(string query)
    {
        DataTable dataTable = new DataTable();
        Debug.WriteLine($"[DEBUG] FetchDataODBCAsync called with query: {query}");

        int attempt = 0;
        int totalRowCount = 0;

        // Retry loop to attempt connecting to the database in case of transient failures
        while (attempt < MaxRetryAttempts)
        {
            try
            {
                // Show Pbar2 as indeterminate before opening connection
                UpdateProgressBar(Pbar2, true, true);

                using (OdbcConnection connection = new OdbcConnection(odbcConnectionString))
                {
                    Debug.WriteLine("[DEBUG] Opening ODBC connection...");
                    await connection.OpenAsync(); // Attempt to open the database connection
                    Debug.WriteLine("[DEBUG] ODBC connection opened successfully.");

                    // Hide Pbar2 immediately after successful connection
                    UpdateProgressBar(Pbar2, false, false);

                    // Prepare count query to get the total number of rows for progress tracking
                    string countQuery = PrepareCountQuery(query);
                    Debug.WriteLine("[DEBUG] Executing count query...");
                    using (OdbcCommand countCommand = new OdbcCommand(countQuery, connection))
                    {
                        // Execute count query to determine the total number of rows
                        totalRowCount = Convert.ToInt32(await countCommand.ExecuteScalarAsync());
                        Debug.WriteLine($"[DEBUG] Total row count: {totalRowCount}");
                    }

                    // Set Pbar3 to determinate mode and set maximum value for tracking progress
                    UpdateProgressBar(Pbar3, true, false);
                    SetProgressBarMaximum(Pbar3, totalRowCount);

                    using (OdbcCommand command = new OdbcCommand(query, connection))
                    {
                        Debug.WriteLine("[DEBUG] Executing ODBC command...");
                        using (var reader = await command.ExecuteReaderAsync(CommandBehavior.SequentialAccess))
                        {
                            if (reader is OdbcDataReader odbcReader)
                            {
                                Debug.WriteLine("[DEBUG] Loading data into DataTable with inferred schema...");

                                // Dynamically create columns in DataTable based on reader schema
                                var schemaTable = odbcReader.GetSchemaTable();
                                if (schemaTable != null)
                                {
                                    foreach (DataRow schemaRow in schemaTable.Rows)
                                    {
                                        // Extract column name and data type from schema
                                        var columnNameObj = schemaRow["ColumnName"];
                                        var dataTypeObj = schemaRow["DataType"];

                                        if (columnNameObj != null && dataTypeObj != null)
                                        {
                                            string? columnName = columnNameObj.ToString();
                                            Type columnType = (Type)dataTypeObj;
                                            dataTable.Columns.Add(columnName, columnType); // Add column to DataTable
                                        }                                  
                                    }
                                }
                           
                                // Buffer to temporarily store rows before adding them to DataTable
                                List<DataRow> rowBuffer = new List<DataRow>();
                                const int batchSize = 1000; // Batch size for buffering rows
                                int rowCounter = 0;

                                // Manually load rows from the reader
                                while (await odbcReader.ReadAsync())
                                {
                                    // Create a new row in the DataTable
                                    DataRow row = dataTable.NewRow();
                                    for (int i = 0; i < odbcReader.FieldCount; i++)
                                    {
                                        // Assign each value from the reader to the appropriate column in the row
                                        row[i] = odbcReader.GetValue(i);
                                    }

                                    // Add the row to the buffer
                                    rowBuffer.Add(row);

                                    // Add rows in batches to the DataTable
                                    if (rowBuffer.Count >= batchSize)
                                    {
                                        foreach (var bufferedRow in rowBuffer)
                                        {
                                            dataTable.Rows.Add(bufferedRow); // Add buffered rows to DataTable
                                        }
                                        rowBuffer.Clear(); // Clear buffer after adding rows

                                        rowCounter += batchSize;

                                        // Update progress bar to reflect rows processed
                                        UpdateProgressBarValue(Pbar3, rowCounter);

                                        Debug.WriteLine($"[DEBUG] Loaded {rowCounter} rows...");
                                    }
                                }

                                // Add any remaining rows in the buffer
                                if (rowBuffer.Count > 0)
                                {
                                    foreach (var bufferedRow in rowBuffer)
                                    {
                                        dataTable.Rows.Add(bufferedRow); // Add remaining buffered rows to DataTable
                                    }

                                    rowCounter += rowBuffer.Count;

                                    // Update progress bar to reflect rows processed
                                    UpdateProgressBarValue(Pbar3, rowCounter);

                                    Debug.WriteLine($"[DEBUG] Loaded remaining {rowBuffer.Count} rows...");
                                }

                                Debug.WriteLine("[DEBUG] Data loaded into DataTable with inferred schema successfully.");
                            }
                            else
                            {
                                throw new InvalidOperationException("ExecuteReaderAsync did not return an OdbcDataReader.");
                            }
                        }
                    }

                    // Hide Pbar3 when data loading is complete
                    UpdateProgressBar(Pbar3, false, false);
                    break; // Exit retry loop after successful execution
                }
            }
            catch (OdbcException odbcEx)
            {
                attempt++;
                Debug.WriteLine($"[ERROR] ODBC error occurred on attempt {attempt}: {odbcEx.Message}");
                if (attempt < MaxRetryAttempts)
                {
                    // Retry connection after a delay
                    Debug.WriteLine($"[DEBUG] Retrying in {RetryDelayMilliseconds} milliseconds...");
                    await Task.Delay(RetryDelayMilliseconds);
                }
                else
                {
                    // Hide Pbar2 if connection fails after all retries
                    UpdateProgressBar(Pbar2, false, false);
                    throw; // Rethrow exception after maximum attempts
                }
            }
            catch (Exception ex)
            {
                // Hide Pbar2 if an unexpected exception occurs
                UpdateProgressBar(Pbar2, false, false);
                Debug.WriteLine($"[ERROR] An unexpected error occurred: {ex.Message}");
                throw; // Rethrow exception
            }
        }
        return dataTable; // Return the loaded DataTable
    }

    // Helper method to prepare the count query by removing ORDER BY clause and other adjustments
    private string PrepareCountQuery(string originalQuery)
    {
        // Remove any trailing semicolons from the original query
        originalQuery = originalQuery.Trim().TrimEnd(';');

        // Remove ORDER BY clause using regex to avoid counting errors
        originalQuery = Regex.Replace(
            originalQuery,
            @"ORDER\s+BY\s+[\w\W\s\d.,]*",
            string.Empty,
            RegexOptions.IgnoreCase);

        // Wrap the original query to create a count query that counts the total rows
        return $"SELECT COUNT(*) FROM ({originalQuery}) T";
    }

    // Helper method to update the visibility and state of a ProgressBar
    private void UpdateProgressBar(ProgressBar progressBar, bool isVisible, bool isIndeterminate)
    {
        // Use the dispatcher to update UI elements safely from a non-UI thread
        progressBar.Dispatcher.Invoke(() =>
        {
            // Set the visibility of the ProgressBar
            progressBar.Visibility = isVisible ? System.Windows.Visibility.Visible : System.Windows.Visibility.Collapsed;
            // Set the indeterminate state of the ProgressBar
            progressBar.IsIndeterminate = isIndeterminate;
        });
    }

    // Helper method to set the maximum value of a ProgressBar
    private void SetProgressBarMaximum(ProgressBar progressBar, int maximum)
    {
        // Use the dispatcher to update UI elements safely from a non-UI thread
        progressBar.Dispatcher.Invoke(() =>
        {
            progressBar.Maximum = maximum; // Set maximum value for the progress bar
            progressBar.Value = 0; // Reset progress bar value
        });
    }

    // Helper method to update the value of a ProgressBar
    private void UpdateProgressBarValue(ProgressBar progressBar, int value)
    {
        // Use the dispatcher to update UI elements safely from a non-UI thread
        progressBar.Dispatcher.Invoke(() =>
        {
            if (value <= progressBar.Maximum)
                progressBar.Value = value; // Update the progress bar value to reflect current progress
            else
                progressBar.Value = progressBar.Maximum; // Cap the progress bar value if it exceeds maximum
        });
    }
}
